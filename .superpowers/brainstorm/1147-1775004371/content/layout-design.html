关键 CSS 约束规则

/* 防溢出三原则 */
.workspace { overflow: hidden; } /* 父容器截断 */
.sidebar, .inspector { overflow-y: auto; min-width: 0; } /* 子列可滚动 */
* { box-sizing: border-box; min-width: 0; } /* 所有元素收缩 */

/* 拖拽宽度通过 CSS 变量控制 */
:root { --sidebar-w: 220px; --inspector-w: 260px; }
/* JS 拖拽时: document.documentElement.style.setProperty('--sidebar-w', x+'px') */

/* 宽度边界 */
--sidebar-w: clamp(160px, var(--sidebar-w), 360px);
--inspector-w: clamp(200px, var(--inspector-w), 480px);
